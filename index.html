export default { async fetch(request) { const url = new URL(request.url); const code = url.searchParams.get('code'); if (!code) return new Response('missing code', { status: 400 }); const corsHeaders = { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': '*', }; // TWSE 收盤價 API（官方，每日更新，不受休市影響） try { const today = new Date(); // 找最近的交易日（往前推，跳過週末） let d = new Date(today); for (let i = 0; i < 5; i++) { d.setDate(d.getDate() - (i === 0 ? 0 : 1)); const day = d.getDay(); if (day !== 0 && day !== 6) break; // 跳過週末 } const dateStr = `${d.getFullYear()}${String(d.getMonth()+1).padStart(2,'0')}${String(d.getDate()).padStart(2,'0')}`; const twseUrl = `https://www.twse.com.tw/rwd/zh/afterTrading/STOCK_DAY?date=${dateStr}&stockNo=${code}&response=json`; const res = await fetch(twseUrl, { headers: { 'User-Agent': 'Mozilla/5.0', 'Referer': 'https://www.twse.com.tw/' } }); if (res.ok) { const data = await res.json(); if (data.stat === 'OK' && data.data?.length) { const last = data.data[data.data.length - 1]; // 欄位: [日期, 成交股數, 成交金額, 開盤價, 最高價, 最低價, 收盤價, 漲跌價差, 成交筆數] const closeStr = last[6]?.replace(/,/g, ''); const close = parseFloat(closeStr); if (close > 0) { const prevCloseStr = data.data.length >= 2 ? data.data[data.data.length-2][6]?.replace(/,/g,'') : closeStr; const prevClose = parseFloat(prevCloseStr) || close; const change = parseFloat((close - prevClose).toFixed(2)); const changePct = parseFloat(((change / prevClose) * 100).toFixed(2)); return new Response(JSON.stringify({ code, price: close, change, changePct, symbol: code + '.TW', source: 'twse', date: last[0] }), { headers: corsHeaders }); } } } } catch {} // 備用：上櫃股票用 TPEx try { const tpexUrl = `https://www.tpex.org.tw/openapi/v1/tpex_mainboard_daily_close_quotes`; const res = await fetch(tpexUrl, { headers: { 'User-Agent': 'Mozilla/5.0' } }); if (res.ok) { const data = await res.json(); const stock = data.find(s => s.SecuritiesCompanyCode === code); if (stock?.Close) { const close = parseFloat(stock.Close); return new Response(JSON.stringify({ code, price: close, change: 0, changePct: 0, symbol: code + '.TWO', source: 'tpex' }), { headers: corsHeaders }); } } } catch {} return new Response( JSON.stringify({ code, price: null, error: 'market closed or not found' }), { headers: corsHeaders } ); } }
